import scipy.io as sio
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import confusion_matrix, accuracy_score, roc_curve, auc

data = sio.loadmat('creditCardDefault-2.mat', simplify_cells=True)
train = data['train']
test = data['test']

columns = ['id','creditLimit','gender','edu','age','nDelay',
           'billAmt1','billAmt2','billAmt3','billAmt4','billAmt5','billAmt6','default']

train_df = pd.DataFrame(train, columns=columns)
test_df  = pd.DataFrame(test, columns=columns)

X_train = train_df.drop(columns=['id','default'])
y_train = (train_df['default'] == 2).astype(int)
X_test  = test_df.drop(columns=['id','default'])
y_test  = (test_df['default'] == 2).astype(int)

scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test  = scaler.transform(X_test)

log_model = LogisticRegression()
log_model.fit(X_train, y_train)
y_pred_log = log_model.predict(X_test)
y_prob_log = log_model.predict_proba(X_test)[:,1]

rf_model = RandomForestClassifier(random_state=42)
rf_model.fit(X_train, y_train)
y_pred_rf = rf_model.predict(X_test)
y_prob_rf = rf_model.predict_proba(X_test)[:,1]

colors = {0: "#bf92a2", 1: "#6b1a2e"}

cm_log = confusion_matrix(y_test, y_pred_log, labels=[0,1])
cm_rf  = confusion_matrix(y_test, y_pred_rf, labels=[0,1])

fpr_log, tpr_log, _ = roc_curve(y_test, y_prob_log)
auc_log = auc(fpr_log, tpr_log)
fpr_rf, tpr_rf, _ = roc_curve(y_test, y_prob_rf)
auc_rf = auc(fpr_rf, tpr_rf)

fig = plt.figure(constrained_layout=True, figsize=(15,12))
fig.set_facecolor('#fdf0f4')
gs = fig.add_gridspec(3, 3, hspace=0.1)

ax0 = fig.add_subplot(gs[0,0])
ax0.scatter(train_df['creditLimit'], train_df['age'], 
            c=train_df['default'].map(lambda x: colors[1] if x==2 else colors[0]),
            edgecolor='#5a1e35', alpha=0.8)
ax0.set_xlabel('Credit Limit', color="#5a1e35")
ax0.set_ylabel('Age', color="#5a1e35")
ax0.set_title('Credit Limit vs Age', color="#92425e")

ax1 = fig.add_subplot(gs[0,1])
ax1.scatter(train_df['billAmt1'], train_df['billAmt2'], 
            c=train_df['default'].map(lambda x: colors[1] if x==2 else colors[0]),
            edgecolor='#5a1e35', alpha=0.8)
ax1.set_xlabel('BillAmt1', color="#5a1e35")
ax1.set_ylabel('BillAmt2', color="#5a1e35")
ax1.set_title('Bill1 vs Bill2', color="#92425e")

ax2 = fig.add_subplot(gs[0,2])
ax2.scatter(train_df['billAmt3'], train_df['billAmt6'], 
            c=train_df['default'].map(lambda x: colors[1] if x==2 else colors[0]),
            edgecolor='#5a1e35', alpha=0.8)
ax2.set_xlabel('BillAmt3', color="#5a1e35")
ax2.set_ylabel('BillAmt6', color="#5a1e35")
ax2.set_title('Bill3 vs Bill6', color="#92425e")

ax3 = fig.add_subplot(gs[1,:])
ax3.plot(fpr_log, tpr_log, label=f'Logistic Regression (AUC={auc_log:.3f})', color="#92425e")
ax3.plot(fpr_rf, tpr_rf, label=f'Random Forest (AUC={auc_rf:.3f})', color="#d4789a")
ax3.plot([0,1],[0,1],'--', color='#bf92a2')
ax3.set_xlabel('False Positive Rate', color="#5a1e35")
ax3.set_ylabel('True Positive Rate', color="#5a1e35")
ax3.set_title('AUC-ROC', color="#92425e")
ax3.legend()

ax4 = fig.add_subplot(gs[2,0])
sns.heatmap(cm_log, annot=True, fmt='d', 
            cmap=sns.blend_palette(["#f6D2DC", "#92425E"], as_cmap=True),
            cbar=False, xticklabels=[0,1], yticklabels=[0,1],
            linewidths=0.5, linecolor='gray', ax=ax4)
ax4.set_xlabel('Predicted', color="#5a1e35")
ax4.set_ylabel('Actual', color="#5a1e35")
ax4.set_title('Logistic Regression', color="#92425e")
ax4.text(0.5, -0.2, f"Accuracy: {accuracy_score(y_test, y_pred_log):.3f}", 
         ha='center', va='center', fontsize=12, color="#92425e", transform=ax4.transAxes)

ax5 = fig.add_subplot(gs[2,1])
sns.heatmap(cm_rf, annot=True, fmt='d', 
            cmap=sns.blend_palette(["#f6D2DC", "#92425E"], as_cmap=True),
            cbar=False, xticklabels=[0,1], yticklabels=[0,1],
            linewidths=0.5, linecolor='gray', ax=ax5)
ax5.set_xlabel('Predicted', color="#5a1e35")
ax5.set_ylabel('Actual', color="#5a1e35")
ax5.set_title('Random Forest', color="#92425e")
ax5.text(0.5, -0.2, f"Accuracy: {accuracy_score(y_test, y_pred_rf):.3f}", 
         ha='center', va='center', fontsize=12, color="#92425e", transform=ax5.transAxes)

shift_x = 0.16
shift_y = -0.06
